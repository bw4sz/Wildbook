#install docker, git and maven
sudo yum -y install  docker 
sudo yum -y install  git 

#install maven for build
sudo wget http://repos.fedorapeople.org/repos/dchen/apache-maven/epel-apache-maven.repo -O /etc/yum.repos.d/epel-apache-maven.repo
sudo sed -i s/\$releasever/6/g /etc/yum.repos.d/epel-apache-maven.repo
sudo yum install -y apache-maven

#install java
sudo yum install -y install java-1.7.0-openjdk

#install tomcat
sudo yum install tomcat6 -y

#git config --global user.name "bw4sz"
#git config --global user.email "benweinstein2010@gmail.com"

git clone https://github.com/bw4sz/Wildbook.git --depth 1 -b 6.x

cd Wildbook

#build war
mvn clean install

#start tomcat
#default page
sudo yum install tomcat6-webapps tomcat6-docs-webapp tomcat6-admin-webapps -Y

#move war
sudo mv target/wildbook-6.0.0-EXPERIMENTAL.war /usr/share/tomcat6/webapps/

sudo service tomcat6 start
